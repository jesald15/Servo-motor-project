#include <Servo.h>

Servo servo1;
Servo servo2;
Servo servo3;

void setup() {
  servo1.attach(22);
  servo2.attach(24);
  servo3.attach(26);
}

void loop() {

  // Move 0 → 180
  for (int pos = 0; pos <= 180; pos++) {
    servo1.write(pos);
    servo2.write(pos);
    servo3.write(pos);
    delay(15);   // speed control (smaller = faster)
  }

  delay(500);  // pause at 180

  // Move 180 → 0
  for (int pos = 180; pos >= 0; pos--) {
    servo1.write(pos);
    servo2.write(pos);
    servo3.write(pos);
    delay(15);
  }

  delay(500);  // pause at 0
}
